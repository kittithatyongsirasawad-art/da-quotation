📊 Google Sheets Integration

Connect to Google Sheets to save quotation records and custom catalog items across all team members. Uses Google Apps Script as a free backend.

Setup Steps (one-time, ~5 minutes):
1. Go to Google Sheets → create a new spreadsheet
2. Click Extensions → Apps Script
3. Delete all code, paste the script below, click Save
4. Click Deploy → New deployment → Web app
5. Set Execute as: Me   |   Who has access: Anyone
6. Click Deploy, copy the Web App URL, paste below
Apps Script Code
function doGet(e) { var action = e.parameter.action || 'ping'; var callback = e.parameter.callback || ''; var ss = SpreadsheetApp.getActiveSpreadsheet(); var result = {}; try { // ── PING ────────────────────────────────────────────────────── if (action === 'ping') { result = { ok: true }; // ── GET ALL RECORDS (summary list) ──────────────────────────── } else if (action === 'get_records') { var sh = ss.getSheetByName('QuotationRecords'); if (!sh) { result = { records: [] }; } else { var rows = sh.getDataRange().getValues().slice(1); result = { records: rows.map(function(r) { return { id: String(r[0] || ''), quotNo: String(r[1] || ''), date: String(r[2] || ''), client: String(r[3] || ''), pkgCount: Number(r[4] || 0), totalRevenue:Number(r[5] || 0), margin: String(r[6] || ''), status: String(r[7] || 'Draft'), preparedBy: String(r[8] || ''), savedAt: String(r[9] || '') }; })}; } // ── GET ONE RECORD DETAIL ───────────────────────────────────── } else if (action === 'get_record_detail') { var id = String(e.parameter.id || ''); var detail = null; // Look in RecordDetails sheet var rd = ss.getSheetByName('RecordDetails'); if (rd && rd.getLastRow() > 1) { var rdRows = rd.getDataRange().getValues().slice(1); for (var i = 0; i < rdRows.length; i++) { if (String(rdRows[i][0]) === id && rdRows[i][1]) { detail = String(rdRows[i][1]); break; } } } result = { ok: true, detail: detail }; // ── SAVE RECORD SUMMARY ─────────────────────────────────────── } else if (action === 'save_record') { var d = JSON.parse(decodeURIComponent(e.parameter.payload || '{}')); var sh = ss.getSheetByName('QuotationRecords'); if (!sh) { sh = ss.insertSheet('QuotationRecords'); sh.appendRow(['ID','QuotNo','Date','Client','Packages','TotalRevenue', 'Margin','Status','PreparedBy','SavedAt']); sh.setFrozenRows(1); sh.getRange('A1:J1').setFontWeight('bold'); sh.setColumnWidth(1, 150); sh.setColumnWidth(2, 130); sh.setColumnWidth(4, 200); sh.setColumnWidth(6, 120); } var newRow = [ String(d.id || ''), String(d.quotNo || ''), String(d.date || ''), String(d.client || ''), Number(d.pkgCount || 0), Number(d.totalRevenue || 0), String(d.margin || ''), String(d.status || 'Draft'), String(d.preparedBy|| ''), new Date().toISOString() ]; // Find existing row to update, or append new var found = false; if (sh.getLastRow() > 1) { var existing = sh.getDataRange().getValues().slice(1); for (var i = 0; i < existing.length; i++) { if (String(existing[i][0]) === String(d.id)) { sh.getRange(i + 2, 1, 1, newRow.length).setValues([newRow]); found = true; break; } } } if (!found) sh.appendRow(newRow); result = { ok: true }; // ── SAVE RECORD DETAIL (full JSON) ──────────────────────────── } else if (action === 'save_record_detail') { // Receives compact detail JSON in one call via payload var d = JSON.parse(decodeURIComponent(e.parameter.payload || '{}')); var id = String(d.id || ''); var detail = String(d.detail || ''); if (!id || !detail) { result = { ok: false, error: 'Missing id or detail' }; } else { var sh = ss.getSheetByName('RecordDetails'); if (!sh) { sh = ss.insertSheet('RecordDetails'); sh.appendRow(['ID','FullJSON','UpdatedAt']); sh.setFrozenRows(1); sh.getRange('A1:C1').setFontWeight('bold'); sh.setColumnWidth(1, 150); sh.setColumnWidth(2, 500); } var now = new Date().toISOString(); var rows = sh.getLastRow() > 1 ? sh.getDataRange().getValues().slice(1) : []; var ri = -1; for (var i = 0; i < rows.length; i++) { if (String(rows[i][0]) === id) { ri = i; break; } } if (ri >= 0) { sh.getRange(ri + 2, 2).setValue(detail); sh.getRange(ri + 2, 3).setValue(now); } else { sh.appendRow([id, detail, now]); } result = { ok: true }; } } else if (action === 'save_catalog_item') { var d = JSON.parse(decodeURIComponent(e.parameter.payload || '{}')); var sh = ss.getSheetByName('CustomCatalog'); if (!sh) { sh = ss.insertSheet('CustomCatalog'); sh.appendRow(['Code','Name','Description','CostNC','Provider', 'IsLump','LumpThreshold','LumpCostNC','AddedAt']); sh.setFrozenRows(1); sh.getRange('A1:I1').setFontWeight('bold'); } sh.appendRow([ String(d.code || ''), String(d.name || ''), String(d.desc || ''), Number(d.costNC || 0), String(d.provider || 'MLL'), d.isLump ? 'TRUE' : 'FALSE', d.lumpThreshold ? Number(d.lumpThreshold) : '', d.lumpCostNC ? Number(d.lumpCostNC) : '', new Date().toISOString() ]); result = { ok: true }; // ── GET CUSTOM CATALOG ──────────────────────────────────────── } else if (action === 'get_catalog') { var sh = ss.getSheetByName('CustomCatalog'); if (!sh) { result = { items: [] }; } else { var rows = sh.getDataRange().getValues().slice(1); result = { items: rows.filter(function(r){ return r[0]; }).map(function(r){ return { code: String(r[0] || ''), name: String(r[1] || ''), desc: String(r[2] || ''), costNC: Number(r[3] || 0), provider: String(r[4] || 'MLL'), isLump: r[5] === 'TRUE' || r[5] === true, lumpThreshold:r[6] ? Number(r[6]) : null, lumpCostNC: r[7] ? Number(r[7]) : null }; })}; } // ── UPDATE STATUS ───────────────────────────────────────────── } else if (action === 'update_status') { var d = JSON.parse(decodeURIComponent(e.parameter.payload || '{}')); var sh = ss.getSheetByName('QuotationRecords'); if (sh && sh.getLastRow() > 1) { var rows = sh.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(d.id)) { sh.getRange(i + 1, 8).setValue(String(d.status || 'Draft')); break; } } } result = { ok: true }; // ── DELETE RECORD ───────────────────────────────────────────── } else if (action === 'delete_record') { var d = JSON.parse(decodeURIComponent(e.parameter.payload || '{}')); var id = String(d.id || ''); var sh = ss.getSheetByName('QuotationRecords'); if (sh && sh.getLastRow() > 1) { var rows = sh.getDataRange().getValues(); for (var i = rows.length - 1; i >= 1; i--) { if (String(rows[i][0]) === id) { sh.deleteRow(i + 1); break; } } } var rd = ss.getSheetByName('RecordDetails'); if (rd && rd.getLastRow() > 1) { var rows = rd.getDataRange().getValues(); for (var i = rows.length - 1; i >= 1; i--) { if (String(rows[i][0]) === id) { rd.deleteRow(i + 1); break; } } } result = { ok: true }; } else { result = { ok: false, error: 'Unknown action: ' + action }; } } catch(err) { result = { ok: false, error: err.message || String(err) }; } var json = JSON.stringify(result); if (callback) { return ContentService.createTextOutput(callback + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService.createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); }
